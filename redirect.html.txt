<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Redirecting...</title>
</head>
<body>
    <h1>Redirecting...</h1>
    <script>
        // Extract the authorization code from the query parameters
        const urlParams = new URLSearchParams(window.location.search);
        const authorizationCode = urlParams.get('code');
        const state = urlParams.get('state');

        if (authorizationCode) {
            // Redirect to your app's custom URI scheme with the authorization code
            window.location.href = `socialjobsapp://linkedin_redirect?code=${authorizationCode}&state=${state}`;
        } else {
            document.body.innerHTML = "<h2>Error: Authorization code not found.</h2>";
        }
    </script>
</body>
</html>
